consider us in a sub apl if:
    the hunt window is in progress
    or elysian decree window is in progress
    or soul carver without fiery demise window is in progress
    or fiery demise with soul carver window is in progress
    or fiery demise without soul carver window is in progress

actions+=/variable,name=sub_apl_in_progress,value=
    variable.the_hunt_ramp_in_progress
    |variable.elysian_decree_ramp_in_progress
    |variable.soul_carver_ramp_in_progress
    |variable.fiery_demise_with_soul_carver_in_progress
    |variable.fiery_demise_without_soul_carver_in_progress

actions+=/variable,name=sub_apl_in_progress,value=variable.the_hunt_ramp_in_progress|variable.elysian_decree_ramp_in_progress|variable.soul_carver_ramp_in_progress|variable.fiery_demise_with_soul_carver_in_progress|variable.fiery_demise_without_soul_carver_in_progress


consider us in a the hunt window if:
    the hunt is talented and
        the hunt is off cooldown within 10s and
            we are not already in a window
consider us not in a the hunt window if:
    the hunt is talented and
        the hunt is on cooldown longer than 10s

actions+=/variable,name=the_hunt_ramp_in_progress,value=1,if=
    talent.the_hunt.enabled
        &cooldown.the_hunt.remains<=10
            &!variable.sub_apl_in_progress
actions+=/variable,name=the_hunt_ramp_in_progress,value=0,if=
    talent.the_hunt.enabled
        &cooldown.the_hunt.remains>10

actions+=/variable,name=the_hunt_ramp_in_progress,value=1,if=talent.the_hunt.enabled&cooldown.the_hunt.remains<=10&!variable.sub_apl_in_progress
actions+=/variable,name=the_hunt_ramp_in_progress,value=0,if=talent.the_hunt.enabled&cooldown.the_hunt.remains>10

actions+=/variable,name=elysian_decree_ramp_in_progress,value=0,if=talent.elysian_decree.enabled&cooldown.elysian_decree.remains>10


consider us in a soul carver without fiery demise window if:
    soul carver is talented and
        fiery demise is not talented and
            soul carver is off cooldown within 10s and
                we are not already in a window
consider us not in a soul carver without fiery demise window if:
    soul carver is talented and
        fiery demise is not talented and
            soul carver is on cooldown longer than 10s

        &!talent.fiery_demise.enabled
            &cooldown.soul_carver.remains<=10
                &!variable.sub_apl_in_progress
actions+=/variable,name=soul_carver_ramp_in_progress,value=0,if=
    talent.soul_carver.enabled
        &!talent.fiery_demise.enabled
            &cooldown.soul_carver.remains>10

actions+=/variable,name=soul_carver_ramp_in_progress,value=1,if=talent.soul_carver.enabled&!talent.fiery_demise.enabled&cooldown.soul_carver.remains<=10&!variable.sub_apl_in_progress
actions+=/variable,name=soul_carver_ramp_in_progress,value=0,if=talent.soul_carver.enabled&!talent.fiery_demise.enabled&cooldown.soul_carver.remains>10


consider us in a fiery demise window if:
    fiery brand is talented and
         fiery demise is talented and
            fiery brand has one or more charges available and
                immolation aura is off cooldown within 2s and
                    we are not already in a window and
                        fel devastation is talented and
                            fel devastation is off cooldown within 10s
                        or soul carver is talented and
                            soul carver is off cooldown within 10s
consider us not in a fiery demise window if:
    fiery brand is talented and
         fiery demise is talented and
            fiery brand has one or less charges available and
                fel devastation is talented and
                    fel devastation is on cooldown longer than 10s
                or soul carver is talented and
                    soul carver is on cooldown longer than 10s

actions+=/variable,name=fiery_demise_in_progress,value=1,if=
    talent.fiery_brand.enabled
        &talent.fiery_demise.enabled
            &cooldown.fiery_brand.charges_fractional>=1
                &cooldown.immolation_aura.remains<=2
                    &!variable.sub_apl_in_progress
                        &(
                            (talent.fel_devastation.enabled
                                &cooldown.fel_devastation.remains_expected<=10)
                            |(talent.soul_carver.enabled
                                &cooldown.soul_carver.remains<=10))
actions+=/variable,name=fiery_demise_in_progress,value=0,if=
    talent.fiery_brand.enabled
        &talent.fiery_demise.enabled
            &cooldown.fiery_brand.charges_fractional<1.65
                &(
                    (talent.fel_devastation.enabled
                        &cooldown.fel_devastation.remains_expected>10)
                    |(talent.soul_carver.enabled
                        &cooldown.soul_carver.remains>10))

actions+=/variable,name=fiery_demise_in_progress,value=1,if=talent.fiery_brand.enabled&talent.fiery_demise.enabled&cooldown.fiery_brand.charges_fractional>=1&cooldown.immolation_aura.remains<=2&!variable.sub_apl_in_progress&((talent.fel_devastation.enabled&cooldown.fel_devastation.remains_expected<=10)|(talent.soul_carver.enabled&cooldown.soul_carver.remains<=10))
actions+=/variable,name=fiery_demise_in_progress,value=0,if=talent.fiery_brand.enabled&talent.fiery_demise.enabled&cooldown.fiery_brand.charges_fraction<1.65&((talent.fel_devastation.enabled&cooldown.fel_devastation.remains_expected>10)|(talent.soul_carver.enabled&cooldown.soul_carver.remains>10))

# The Hunt APL #
################

before combat, define "spirit bomb soul fragments not in meta requirement" as:
    4 if fracture is talented
    5 otherwise
before combat, define "spirit bomb soul fragments in meta requirement" as:
    3 if fracture is talented
    4 otherwise

define "spirit bomb soul fragments requirement" as:
    "spirit bomb soul fragments in meta requirement" if metamorphosis is active
    "spirit bomb soul fragments not in meta requirement" otherwise
define "frailty target requirement" as:
    5 if:
        spirit bomb is talented and
            spirit bomb can hit more than 1 target
        or soul cleave can hit more than 1 target
    6 otherwise
define "frailty dump fury requirement" as:
    1 SpB fury cost + 2 SC fury cost if:
        spirit bomb is talented and
            spirit bomb can hit more than 1 target
        or soul cleave can hit more than 1 target
    3 SC fury cost otherwise
define "pooling for the hunt" as:
    true if:
        the hunt is talented and
            the hunt comes off cooldown within 2 gcds and
                available fury is less than "frailty dump fury requirement" and
                    target has less than or equal to 1 stack of frailty
    false otherwise
define "pooling fury" as:
    true if:
        "pooling for the hunt"
    false otherwise
cast the hunt if:
    target has greater than or equal to "frailty target requirement"
cast spirit bomb if:
    not "pooling fury" and
        available soul fragments is greater than or equal to "spirit bomb soul fragments requirement" and
            spirit bomb can hit more than 1 target
cast soul cleave if:
    not "pooling fury" and
        available soul fragments are less than or equal to 1 and
            soul cleave can hit more than 1 target
        or soul cleave can hit less than 2 targets
cast sigil of flame
cast fracture
cast shear
cast throw glaive
cast felblade

    "spirit bomb soul fragments in meta requirement" if metamorphosis is active
    "spirit bomb soul fragments not in meta requirement" otherwise

define "frailty target requirement" as:
    5 if:
        spirit bomb is talented and
            spirit bomb can hit more than 1 target
        or soul cleave can hit more than 1 target
    6 otherwise

action+=/variable,name=frailty_target_requirement,op=setif,value=5,value_else=6,condition=spell_targets.spirit_bomb>1|spell_targets.soul_cleave>1


define "frailty dump fury requirement" as:
    1 SpB fury cost + 2 SC fury cost if:
        spirit bomb is talented and
            spirit bomb can hit more than 1 target
        or soul cleave can hit more than 1 target
    3 SC fury cost otherwise

actions+=/variable,name=frailty_dump_fury_requirement,op=setif,value=action.spirit_bomb.cost+(action.soul_cleave.cost*2),value_else=action.soul_cleave.cost*3,condition=spell_targets.spirit_bomb>1|spell_targets.soul_cleave


define "pooling for the hunt" as:
    true if:
        the hunt is talented and
            the hunt comes off cooldown within 2 gcds and
                available fury is less than "frailty dump fury requirement" and
                    target has less than or equal to 1 stack of frailty
    false otherwise

actions+=/variable,name=pooling_for_the_hunt,value=
    talent.the_hunt.enabled
        &cooldown.the_hunt.remains<(gcd.max*2)
            &fury<variable.frailty_dump_fury_requirement
                &debuff.frailty.stack<=1

actions+=/variable,name=pooling_for_the_hunt,value=talent.the_hunt.enabled&cooldown.the_hunt.remains<(gcd.max*2)&fury<variable.frailty_dump_fury_requirement&debuff.frailty.stack<=1


define "pooling fury" as:
    true if:
        "pooling for the hunt"
    false otherwise

actions+=/variable,name=pooling_for_the_hunt,value=
    variable.pooling_for_the_hunt

actions+=/variable,name=pooling_for_the_hunt,value=variable.pooling_for_the_hunt


cast the hunt if:
    target has greater than or equal to "frailty target requirement"

actions.the_hunt_ramp=the_hunt,if=
    debuff.frailty.stack>=variable.frailty_target_requirement

actions.the_hunt_ramp=the_hunt,if=debuff.frailty.stack>=variable.frailty_target_requirement


cast spirit bomb if:
    not "pooling fury" and
        available soul fragments is greater than or equal to "spirit bomb soul fragments requirement" and
            spirit bomb can hit more than 1 target

actions.the_hunt_ramp+=/spirit_bomb,if=
    !variable.pooling_fury
        &soul_fragments>=variable.spirit_bomb_soul_fragments
            &spell_targets>1

actions.the_hunt_ramp+=/spirit_bomb,if=!variable.pooling_fury&soul_fragments>=variable.spirit_bomb_soul_fragments&spell_targets>1


cast soul cleave if:
    not "pooling fury" and
        available soul fragments are less than or equal to 1 and
            soul cleave can hit more than 1 target
        or soul cleave can hit less than 2 targets

actions.the_hunt_ramp+=/soul_cleave,if=
    !variable.pooling_fury
        &(
            (soul_fragments<=1
                &spell_targets>1)
            |spell_targets<2)


cast sigil of flame

actions.the_hunt_ramp+=/sigil_of_flame


cast fracture

actions.the_hunt_ramp+=/fracture


cast shear

actions.the_hunt_ramp+=/shear
